About Me

Crafting Worlds,

I'm Nikhil Srivastwa, a Lead VFX Compositor with over 12 years of experience delivering high-end visual effects for feature films, episodic television, and commercial productions. # Smart Grade Gizmo — auto-balances # shadows, mids, and highs via luma key import nuke def create_smart_grade(): g = nuke.createNode('Group') g['name'].setValue('NS_SmartGrade') with g: inp = nuke.createNode('Input') grade = nuke.createNode('Grade') grade.setInput(0, inp) return g

# Auto-Write: generates versioned # output paths from shot metadata import nuke, os, re def auto_write_path(node): root = nuke.root() name = root['name'].value() ver = re.findall( r'v(\d+)', name) v = ver[-1] if ver else '001' return f'/renders/v{v}/####.exr'
